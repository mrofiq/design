12. Usage Examples

HTML Usage

<!-- Basic Button -->
<button class="btn btn-primary">Click Me</button>

<!-- Button with Icon -->
<button class="btn btn-primary">
  <svg class="icon">...</svg>
  <span>Submit</span>
</button>

<!-- Loading Button -->
<button class="btn btn-primary is-loading">Processing...</button>

<!-- Auto-initialize with JavaScript -->
<button class="btn btn-primary" data-button='{"ripple": true}'>
  Auto Init
</button>

JavaScript Usage

// Create button instance
const myButton = new Button('#myButton', {
  ripple: true,
  loadingText: 'Processing...'
});

// Set loading state
myButton.startLoading('Submitting...');
setTimeout(() => myButton.stopLoading(), 2000);

// Enable/Disable
myButton.disable();
myButton.enable();

// Listen to events
myButton.on('button:click', (event) => {
  console.log('Button clicked!', event.detail);
});
