Usage Examples

Basic Modal

const modal = new Modal({
  size: 'md',
  closeOnBackdrop: true,
  closeOnEscape: true
});

modal.setContent({
  title: 'Modal Title',
  body: 'Modal content goes here.',
  footer: '<button class="btn btn-primary" data-modal-close>Close</button>',
  showClose: true
});

modal.open();

Confirmation Helper

const confirmed = await ModalHelpers.confirm({
  title: 'Confirm Action',
  message: 'Are you sure you want to proceed?',
  confirmText: 'Yes, Proceed',
  cancelText: 'Cancel'
});

if (confirmed) {
  // User clicked confirm
}

Alert Helper

await ModalHelpers.success('Operation completed successfully!');
await ModalHelpers.warning('Please review your changes.');
await ModalHelpers.error('An error occurred.');
await ModalHelpers.info('For your information.');
